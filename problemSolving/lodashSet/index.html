Problem Statement:

Implement a method in Javascript that will take an object and a string or array of strings as a path and set the value at that path.

Example:

const obj = {
    a: {
        b: {
            c: [1, 2, 3]
        }
    }
};

console.log(set(obj, 'a.b.c', 4));      // { a: { b: { c: 4 } } }
console.log(set(obj, 'a.b.c.0', 4));    // { a: { b: { c: [4, 2, 3] } } }
console.log(set(obj, 'a.b.c[1]', 4));   // { a: { b: { c: [4, 4, 3] } } }
console.log(set(obj, 'a.b.c[3]', 4));   // { a: { b: { c: [4, 4, 3, 4] } } }
